## Overview

v0.2.0 delivered the frontend scaffold: React 19, TypeScript, Tailwind 4, TanStack Router/Query, Zustand, shell layout, API client (`src/lib/api.ts`), shared components, and basic route stubs. The current Dashboard (`src/routes/index.tsx`) shows a project list with minimal stats (project count, active sprints, total sprints) and simple project cards.

v0.2.1 transforms this into a functional project management layer:
- **Rich Dashboard** with analytics-backed stats and actionable project cards
- **3-step Project Creation Wizard** with Zod-validated forms
- **Backend `POST /api/projects/init`** that creates full project structure on disk
- **Project Detail page** with tabbed navigation
- **Team Config page** for managing agents
- **Template Browser** with filtering and wizard integration

### Tech Decisions

| Decision | Choice | Rationale |
|----------|--------|-----------|
| Form library | React Hook Form + Zod | Type-safe validation, wizard-friendly |
| Step state | Zustand store | Persists across wizard steps, survives re-renders |
| Inline editing | Controlled input with blur-save | Simple, no extra library |
| Template filtering | Client-side | Template list is small (<100), no need for server-side |
| Tab routing | URL search params `?tab=sprints` | Shareable links, back button works |

### Data Fetching Strategy

```typescript
// src/hooks/use-dashboard-stats.ts
export function useDashboardStats() {
  const projectsQuery = useProjects();
  
  // For each project, fetch cost + agents in parallel
  const projects = projectsQuery.data?.projects ?? [];
  
  const enrichedProjects = useQueries({
    queries: projects.map((p) => ({
      queryKey: ['project-dashboard', p.project_id],
      queryFn: async () => {
        const [agents, cost, sprints] = await Promise.all([
          api.agents.list(p.project_id),
          api.analytics.projectCost(p.project_id).catch(() => ({ total_cost_usd: 0 })),
          api.sprints.list(p.project_id),
        ]);
        return {
          project: p,
          agentCount: agents.agents.length,
          totalCost: cost.total_cost_usd,
          lastSprint: sprints.sprints[0] ?? null,
        };
      },
      enabled: !!p.project_id,
      staleTime: 30_000, // 30s — dashboard isn't realtime-critical
    })),
  });

  const aggregated = {
    totalProjects: projects.length,
    activeSprints: projects.filter((p) => p.current_sprint_id).length,
    totalCost: enrichedProjects.reduce(
      (sum, q) => sum + (q.data?.totalCost ?? 0), 0
    ),
    totalLearnings: 0, // fetched separately or from a dedicated endpoint
  };

  return { projects: enrichedProjects, stats: aggregated, isLoading: projectsQuery.isLoading };
}
```

```python
class AgentInitConfig(BaseModel):
    role: str = Field(..., min_length=1)
```

### Implementation

```python
@router.post("/projects/init", status_code=201)
async def init_project(body: ProjectInitRequest) -> ProjectInitResponse:
    """Initialize a full project from the wizard."""
    import os
    import yaml

    db = await get_db()
    project_id = str(uuid.uuid4())
    work_dir = os.path.expanduser(body.working_directory)

    # 1. Create directory structure
    os.makedirs(work_dir, exist_ok=True)
    foundrai_dir = os.path.join(work_dir, ".foundrai")
    os.makedirs(foundrai_dir, exist_ok=True)

    # 2. Generate foundrai.yaml
    config = {
        "project": {
            "name": body.name,
            "description": body.description,
            "id": project_id,
        },
        "team": {
            agent.role: {
                "model": agent.model,
                "autonomy": agent.autonomy,
                "enabled": agent.enabled,
                **({"system_prompt": agent.system_prompt} if agent.system_prompt else {}),
            }
            for agent in body.agents
        },
        "sprint": {
            "max_parallel_tasks": body.sprint_defaults.max_parallel_tasks,
            "token_budget_usd": body.sprint_defaults.token_budget_usd,
            "sandbox": body.sprint_defaults.sandbox_provider,
        },
    }
    config_path = os.path.join(work_dir, "foundrai.yaml")
    with open(config_path, "w") as f:
        yaml.dump(config, f, default_flow_style=False, sort_keys=False)

    # 3. Create .gitignore
    gitignore_path = os.path.join(work_dir, ".gitignore")
    if not os.path.exists(gitignore_path):
        with open(gitignore_path, "w") as f:
            f.write(".foundrai/\n.env\n__pycache__/\nnode_modules/\n")

    # 4. Insert project into DB
    now = datetime.utcnow().isoformat()
    await db.conn.execute(
        "INSERT INTO projects (project_id, name, description, created_at, working_directory) VALUES (?, ?, ?, ?, ?)",
        (project_id, body.name, body.description, now, work_dir),
    )

    # 5. Insert agent configs into DB
    for agent in body.agents:
        await db.conn.execute(
            "INSERT INTO agent_configs (project_id, agent_role, autonomy_level, model, enabled) VALUES (?, ?, ?, ?, ?)",
            (project_id, agent.role, agent.autonomy, agent.model, int(agent.enabled)),
        )

    await db.conn.commit()

    return ProjectInitResponse(
        project_id=project_id,
        name=body.name,
        working_directory=work_dir,
        config_path=config_path,
        created_at=now,
    )
```

### Inline Editing Implementation

```typescript
// InlineEditableText — simplified logic
function InlineEditableText({ value, onSave, as: Tag = 'p', placeholder, maxLength }: InlineEditableTextProps) {
  const [editing, setEditing] = useState(false);
  const [draft, setDraft] = useState(value);

  const handleBlur = () => {
    setEditing(false);
    if (draft.trim() !== value) {
      onSave(draft.trim());
    }
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') (e.target as HTMLElement).blur();
    if (e.key === 'Escape') { setDraft(value); setEditing(false); }
  };

  if (!editing) {
    return <Tag onClick={() => setEditing(true)} className="cursor-pointer hover:bg-muted/20 rounded px-1 -mx-1">{value || placeholder}</Tag>;
  }

  return <input value={draft} onChange={(e) => setDraft(e.target.value)} onBlur={handleBlur} onKeyDown={handleKeyDown} maxLength={maxLength} autoFocus className="bg-transparent border-b border-primary outline-none" />;
}
```

### Backend: Project Update Endpoint (new)

```
PATCH /api/projects/{project_id}
Body: { name?: string, description?: string }
Returns: ProjectResponse
```

```python
class ProjectUpdate(BaseModel):
    name: str | None = None
    description: str | None = None

@router.patch("/projects/{project_id}")
async def update_project(project_id: str, body: ProjectUpdate) -> ProjectResponse:
    db = await get_db()
    updates, params = [], []
    if body.name is not None:
        updates.append("name = ?"); params.append(body.name)
    if body.description is not None:
        updates.append("description = ?"); params.append(body.description)
    if not updates:
        raise HTTPException(400, "No fields to update")
    params.append(project_id)
    await db.conn.execute(f"UPDATE projects SET {', '.join(updates)} WHERE project_id = ?", params)
    await db.conn.commit()
    return await get_project(project_id)
```

### Save Logic

The team config page uses local state for editing, then saves all changes at once:

```typescript
// src/hooks/use-team-config.ts
export function useTeamConfig(projectId: string) {
  const queryClient = useQueryClient();
  const agentsQuery = useQuery({
    queryKey: ['agents', projectId],
    queryFn: () => api.agents.list(projectId),
  });

  const [localAgents, setLocalAgents] = useState<AgentConfig[]>([]);
  const [isDirty, setIsDirty] = useState(false);

  // Sync from server on first load
  useEffect(() => {
    if (agentsQuery.data) {
      setLocalAgents(agentsQuery.data.agents);
      setIsDirty(false);
    }
  }, [agentsQuery.data]);

  const updateAgent = (index: number, updates: Partial<AgentConfig>) => {
    setLocalAgents((prev) => prev.map((a, i) => i === index ? { ...a, ...updates } : a));
    setIsDirty(true);
  };

  const addAgent = (agent: AgentConfig) => {
    setLocalAgents((prev) => [...prev, agent]);
    setIsDirty(true);
  };

  const removeAgent = (index: number) => {
    setLocalAgents((prev) => prev.filter((_, i) => i !== index));
    setIsDirty(true);
  };

  const saveMutation = useMutation({
    mutationFn: async () => {
      // Save each agent via PUT (upsert)
      await Promise.all(
        localAgents.map((a) =>
          api.agents.update(projectId, a.agent_role, {
            autonomy_level: a.autonomy_level,
            model: a.model,
            enabled: a.enabled,
          })
        )
      );
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['agents', projectId] });
      setIsDirty(false);
    },
  });

  return {
    agents: localAgents,
    isDirty,
    isLoading: agentsQuery.isLoading,
    updateAgent,
    addAgent,
    removeAgent,
    save: saveMutation.mutate,
    isSaving: saveMutation.isPending,
    discard: () => {
      setLocalAgents(agentsQuery.data?.agents ?? []);
      setIsDirty(false);
    },
  };
}
```

### Filter Logic (client-side)

```typescript
// src/hooks/use-templates.ts
export function useFilteredTemplates() {
  const templatesQuery = useQuery({
    queryKey: ['templates'],
    queryFn: () => api.templates.list(),
  });

  const [search, setSearch] = useState('');
  const [selectedType, setSelectedType] = useState('all');
  const [selectedTags, setSelectedTags] = useState<string[]>([]);

  const allTemplates = templatesQuery.data ?? [];
  
  const availableTags = useMemo(
    () => [...new Set(allTemplates.flatMap((t) => t.tags))].sort(),
    [allTemplates]
  );

  const filtered = useMemo(() => {
    return allTemplates.filter((t) => {
      if (search && !t.name.toLowerCase().includes(search.toLowerCase()) &&
          !t.description.toLowerCase().includes(search.toLowerCase())) return false;
      if (selectedType !== 'all' && !t.tags.includes(selectedType)) return false;
      if (selectedTags.length > 0 && !selectedTags.some((tag) => t.tags.includes(tag))) return false;
      return true;
    });
  }, [allTemplates, search, selectedType, selectedTags]);

  return {
    templates: filtered,
    allTemplates,
    isLoading: templatesQuery.isLoading,
    search, setSearch,
    selectedType, setSelectedType,
    selectedTags, setSelectedTags,
    availableTags,
  };
}
```

```typescript
// src/lib/schemas.ts
import { z } from 'zod';

// === Step 1: Basics ===
export const wizardStep1Schema = z.object({
  name: z
    .string()
    .min(1, 'Project name is required')
    .max(100, 'Project name must be 100 characters or less')
    .regex(/^[a-zA-Z0-9\s\-_]+$/, 'Only letters, numbers, spaces, hyphens, and underscores'),
  description: z
    .string()
    .max(500, 'Description must be 500 characters or less')
    .default(''),
  workingDirectory: z
    .string()
    .min(1, 'Working directory is required')
    .refine(
      (path) => path.startsWith('/') || path.startsWith('~'),
      'Must be an absolute path (starts with / or ~)'
    ),
});

// === Step 2: Agent Config ===
export const agentConfigSchema = z.object({
  role: z.string().min(1, 'Role is required'),
  model: z.string().min(1, 'Model is required'),
  autonomy: z.enum(['low', 'medium', 'high']),
  enabled: z.boolean(),
  systemPrompt: z.string().max(2000).default(''),
});

export const wizardStep2Schema = z.object({
  agents: z
    .array(agentConfigSchema)
    .min(1, 'At least one agent is required')
    .refine(
      (agents) => agents.some((a) => a.role === 'pm' && a.enabled),
      'A PM agent is required'
    )
    .refine(
      (agents) => agents.some((a) => a.role === 'developer' && a.enabled),
      'A Developer agent is required'
    ),
});

// === Step 3: Sprint Defaults ===
export const sprintDefaultsSchema = z.object({
  maxParallelTasks: z.number().int().min(1).max(10).default(3),
  tokenBudgetUsd: z.number().min(0.1).max(100).default(5.0),
  sandboxProvider: z.enum(['docker', 'local', 'none']).default('docker'),
});

export const wizardStep3Schema = z.object({
  sprintDefaults: sprintDefaultsSchema,
});

// === Full Wizard Form ===
export const projectWizardSchema = wizardStep1Schema
  .merge(wizardStep2Schema)
  .merge(wizardStep3Schema);

export type ProjectWizardFormData = z.infer<typeof projectWizardSchema>;

// === Team Config Page (for saving) ===
export const teamConfigSaveSchema = z.object({
  agents: z.array(
    z.object({
      agent_role: z.string().min(1),
      autonomy_level: z.enum(['low', 'medium', 'high']),
      model: z.string().min(1),
      enabled: z.boolean(),
    })
  ).min(1),
});

// === Inline Edit ===
export const projectNameSchema = z.string().min(1).max(100);
export const projectDescriptionSchema = z.string().max(500);
```
