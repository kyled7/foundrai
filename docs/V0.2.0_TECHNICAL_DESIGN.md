### Theme

- **Dark mode by default** — `<html class="dark">` set on load
- Theme toggle persisted to `localStorage`
- CSS variables follow shadcn/ui conventions (`--background`, `--foreground`, `--primary`, etc.)
- Color palette: neutral grays for chrome, blue primary, green/red/amber for status semantics

```ts
// src/lib/theme.ts
export type Theme = "dark" | "light" | "system";

export function getTheme(): Theme {
  return (localStorage.getItem("foundrai-theme") as Theme) ?? "dark";
}

export function setTheme(theme: Theme) {
  localStorage.setItem("foundrai-theme", theme);
  const root = document.documentElement;
  root.classList.remove("light", "dark");
  if (theme === "system") {
    const sys = window.matchMedia("(prefers-color-scheme: dark)").matches ? "dark" : "light";
    root.classList.add(sys);
  } else {
    root.classList.add(theme);
  }
}
```

## 2. Routing

### TanStack Router — File-Based Routes

| Path | Component | Description |
|------|-----------|-------------|
| `/` | `DashboardPage` | Project list, recent sprints, pending approvals |
| `/projects/new` | `ProjectCreatePage` | Project creation wizard |
| `/projects/$id` | `ProjectDetailPage` | Project overview, current sprint summary |
| `/projects/$id/sprint` | `SprintCommandPage` | Sprint command center (live feed) |
| `/projects/$id/sprint/$sid` | `SprintReplayPage` | Sprint replay (historical) |
| `/projects/$id/analytics` | `AnalyticsPage` | Cost, performance, comparisons |
| `/projects/$id/team` | `TeamConfigPage` | Agent config, team management |
| `/templates` | `TemplateBrowserPage` | Browse/create templates |
| `/settings` | `SettingsPage` | Global settings |

### Loaders

Each route that needs data uses TanStack Router loaders integrated with TanStack Query:

```ts
// src/routes/projects/$id/index.tsx
import { createFileRoute } from "@tanstack/react-router";
import { projectQueryOptions } from "@/hooks/useProjects";

export const Route = createFileRoute("/projects/$id/")({
  loader: ({ context: { queryClient }, params: { id } }) =>
    queryClient.ensureQueryData(projectQueryOptions(id)),
  component: ProjectDetailPage,
});
```

### 3.2 API Client

```ts
// src/lib/api.ts

const BASE_URL = import.meta.env.VITE_API_URL ?? "";

export class ApiError extends Error {
  constructor(
    public status: number,
    public detail: string,
    public body?: unknown,
  ) {
    super(detail);
    this.name = "ApiError";
  }
}

async function request<T>(
  path: string,
  options: RequestInit = {},
): Promise<T> {
  const url = `${BASE_URL}${path}`;
  const res = await fetch(url, {
    headers: {
      "Content-Type": "application/json",
      ...options.headers,
    },
    ...options,
  });

  if (!res.ok) {
    const body = await res.json().catch(() => null);
    throw new ApiError(
      res.status,
      body?.detail ?? res.statusText,
      body,
    );
  }

  if (res.status === 204) return undefined as T;
  return res.json();
}

function get<T>(path: string) {
  return request<T>(path);
}

function post<T>(path: string, body?: unknown) {
  return request<T>(path, {
    method: "POST",
    body: body ? JSON.stringify(body) : undefined,
  });
}

function put<T>(path: string, body?: unknown) {
  return request<T>(path, {
    method: "PUT",
    body: body ? JSON.stringify(body) : undefined,
  });
}

// ─── Endpoint Functions ──────────────────────────────────────

import type * as T from "@/types/api";

export const api = {
  // Projects
  projects: {
    list: () => get<T.ProjectListResponse>("/api/projects"),
    get: (id: string) => get<T.ProjectResponse>(`/api/projects/${id}`),
    create: (body: T.ProjectCreate) =>
      post<T.ProjectResponse>("/api/projects", body),
  },

  // Sprints
  sprints: {
    list: (projectId: string) =>
      get<T.SprintListResponse>(`/api/projects/${projectId}/sprints`),
    get: (id: string) => get<T.SprintDetail>(`/api/sprints/${id}`),
    create: (projectId: string, body: T.SprintCreate) =>
      post<T.SprintCreateResponse>(`/api/projects/${projectId}/sprints`, body),
    metrics: (id: string) => get<T.SprintMetrics>(`/api/sprints/${id}/metrics`),
    goalTree: (id: string) =>
      get<T.GoalTreeResponse>(`/api/sprints/${id}/goal-tree`),
  },

  // Tasks
  tasks: {
    list: (sprintId: string) =>
      get<T.TaskDetail[]>(`/api/sprints/${sprintId}/tasks`),
    get: (id: string) => get<T.TaskSummary>(`/api/tasks/${id}`),
  },

  // Agents
  agents: {
    list: (projectId: string) =>
      get<T.AgentListResponse>(`/api/projects/${projectId}/agents`),
    update: (projectId: string, role: string, body: T.AgentConfigUpdate) =>
      put<{ project_id: string; agent_role: string; status: string }>(
        `/api/projects/${projectId}/agents/${role}`,
        body,
      ),
  },

  // Approvals
  approvals: {
    list: (sprintId: string) =>
      get<T.ApprovalListResponse>(`/api/sprints/${sprintId}/approvals`),
    approve: (id: string, body?: T.ApprovalDecision) =>
      post<T.ApprovalResponse>(`/api/approvals/${id}/approve`, body ?? {}),
    reject: (id: string, body?: T.ApprovalDecision) =>
      post<T.ApprovalResponse>(`/api/approvals/${id}/reject`, body ?? {}),
  },

  // Events & Messages
  events: {
    list: (sprintId: string) =>
      get<T.EventListResponse>(`/api/sprints/${sprintId}/events`),
    messages: (sprintId: string) =>
      get<T.MessageListResponse>(`/api/sprints/${sprintId}/messages`),
  },

  // Artifacts
  artifacts: {
    list: (sprintId: string) =>
      get<T.ArtifactListResponse>(`/api/sprints/${sprintId}/artifacts`),
  },

  // Analytics
  analytics: {
    sprintCost: (sprintId: string) =>
      get<T.CostUsage>(`/api/sprints/${sprintId}/cost`),
    projectCost: (projectId: string) =>
      get<T.CostUsage>(`/api/projects/${projectId}/cost`),
    agentCosts: (projectId: string) =>
      get<T.AgentCostResponse>(`/api/projects/${projectId}/agent-costs`),
    budget: (sprintId: string) =>
      get<T.BudgetStatus>(`/api/sprints/${sprintId}/budget`),
    setBudget: (sprintId: string, body: T.BudgetOverrideRequest) =>
      put<{ status: string }>(`/api/sprints/${sprintId}/budget`, body),
    commGraph: (sprintId: string) =>
      get<T.CommGraphResponse>(`/api/sprints/${sprintId}/comm-graph`),
    sprintComparison: (projectId: string) =>
      get<T.SprintComparisonResponse>(
        `/api/projects/${projectId}/sprint-comparison`,
      ),
  },

  // Templates
  templates: {
    list: (params?: { source?: string; author?: string; public_only?: boolean }) => {
      const qs = new URLSearchParams();
      if (params?.source) qs.set("source", params.source);
      if (params?.author) qs.set("author", params.author);
      if (params?.public_only) qs.set("public_only", "true");
      const q = qs.toString();
      return get<T.TeamTemplate[]>(`/api/templates${q ? `?${q}` : ""}`);
    },
    get: (id: string) => get<T.TeamTemplate>(`/api/templates/${id}`),
    create: (body: T.CreateTemplateRequest) =>
      post<T.TeamTemplate>("/api/templates", body),
    apply: (templateId: string, body: T.ApplyTemplateRequest) =>
      post<T.ApplyTemplateResponse>(`/api/templates/${templateId}/apply`, body),
  },

  // Teams
  teams: {
    list: (projectId: string) =>
      get<T.Team[]>(`/api/projects/${projectId}/teams`),
    create: (projectId: string, body: T.CreateTeamRequest) =>
      post<T.Team>(`/api/projects/${projectId}/teams`, body),
  },

  // Learnings
  learnings: {
    list: (projectId: string) =>
      get<T.LearningListResponse>(`/api/projects/${projectId}/learnings`),
  },

  // Replay
  replay: {
    get: (sprintId: string) =>
      get<T.ReplayResponse>(`/api/sprints/${sprintId}/replay`),
  },
} as const;
```

```ts
// src/hooks/useProjects.ts
import { queryOptions, useMutation, useQueryClient } from "@tanstack/react-query";
import { api } from "@/lib/api";
import type { ProjectCreate } from "@/types/api";

export const projectKeys = {
  all: ["projects"] as const,
  detail: (id: string) => ["projects", id] as const,
};

export const projectListOptions = () =>
  queryOptions({
    queryKey: projectKeys.all,
    queryFn: () => api.projects.list(),
  });

export const projectQueryOptions = (id: string) =>
  queryOptions({
    queryKey: projectKeys.detail(id),
    queryFn: () => api.projects.get(id),
  });

export function useCreateProject() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (body: ProjectCreate) => api.projects.create(body),
    onSuccess: () => qc.invalidateQueries({ queryKey: projectKeys.all }),
  });
}
```

### 4.2 WebSocket Client with Auto-Reconnect

```ts
// src/lib/ws.ts

import type { WsEvent } from "@/types/ws";

export type WsEventHandler = (event: WsEvent) => void;

interface WsClientOptions {
  url: string;
  sprintId: string;
  onEvent: WsEventHandler;
  onConnect?: () => void;
  onDisconnect?: () => void;
  maxRetries?: number;
}

export class WsClient {
  private ws: WebSocket | null = null;
  private retryCount = 0;
  private maxRetries: number;
  private lastSeq = 0;
  private disposed = false;
  private reconnectTimer: ReturnType<typeof setTimeout> | null = null;
  private opts: WsClientOptions;

  constructor(opts: WsClientOptions) {
    this.opts = opts;
    this.maxRetries = opts.maxRetries ?? 10;
    this.connect();
  }

  private connect() {
    if (this.disposed) return;

    // Include lastSeq for catch-up on reconnect
    const sep = this.opts.url.includes("?") ? "&" : "?";
    const url = `${this.opts.url}${sep}sprint_id=${this.opts.sprintId}&after_seq=${this.lastSeq}`;

    this.ws = new WebSocket(url);

    this.ws.onopen = () => {
      this.retryCount = 0;
      this.opts.onConnect?.();
    };

    this.ws.onmessage = (msg) => {
      try {
        const event: WsEvent = JSON.parse(msg.data);
        if (event.seq > this.lastSeq) {
          this.lastSeq = event.seq;
        }
        this.opts.onEvent(event);
      } catch {
        // Ignore malformed messages
      }
    };

    this.ws.onclose = () => {
      this.opts.onDisconnect?.();
      this.scheduleReconnect();
    };

    this.ws.onerror = () => {
      this.ws?.close();
    };
  }

  private scheduleReconnect() {
    if (this.disposed || this.retryCount >= this.maxRetries) return;

    // Exponential backoff: 1s, 2s, 4s, 8s, ... capped at 30s
    const delay = Math.min(1000 * 2 ** this.retryCount, 30_000);
    this.retryCount++;

    this.reconnectTimer = setTimeout(() => this.connect(), delay);
  }

  dispose() {
    this.disposed = true;
    if (this.reconnectTimer) clearTimeout(this.reconnectTimer);
    this.ws?.close();
  }
}
```

### 4.3 Zustand Store for Sprint Feed

```ts
// src/stores/sprintFeed.ts

import { create } from "zustand";
import type { WsEvent, WsEventType } from "@/types/ws";
import { WsClient } from "@/lib/ws";

interface SprintFeedState {
  // Connection
  connected: boolean;
  sprintId: string | null;

  // Event log (ring buffer, last 500)
  events: WsEvent[];
  lastSeq: number;

  // Derived counters (updated on each event)
  pendingApprovals: number;

  // Actions
  connect: (sprintId: string) => void;
  disconnect: () => void;
  _pushEvent: (event: WsEvent) => void;
}

const MAX_EVENTS = 500;
const WS_BASE = import.meta.env.VITE_WS_URL ?? `ws://${window.location.host}`;

let client: WsClient | null = null;

export const useSprintFeed = create<SprintFeedState>((set, get) => ({
  connected: false,
  sprintId: null,
  events: [],
  lastSeq: 0,
  pendingApprovals: 0,

  connect: (sprintId: string) => {
    // Tear down existing
    client?.dispose();

    set({ sprintId, events: [], lastSeq: 0 });

    client = new WsClient({
      url: `${WS_BASE}/ws/sprint`,
      sprintId,
      onEvent: (event) => get()._pushEvent(event),
      onConnect: () => set({ connected: true }),
      onDisconnect: () => set({ connected: false }),
    });
  },

  disconnect: () => {
    client?.dispose();
    client = null;
    set({ connected: false, sprintId: null });
  },

  _pushEvent: (event) => {
    set((state) => {
      const events =
        state.events.length >= MAX_EVENTS
          ? [...state.events.slice(-MAX_EVENTS + 1), event]
          : [...state.events, event];

      let { pendingApprovals } = state;
      if (event.type === "approval.requested") pendingApprovals++;
      if (event.type === "approval.approved" || event.type === "approval.rejected")
        pendingApprovals = Math.max(0, pendingApprovals - 1);

      return { events, lastSeq: event.seq, pendingApprovals };
    });
  },
}));
```

```python
# FoundrAI/api/spa.py  (added in v0.2.0)
from pathlib import Path
from fastapi import FastAPI
from fastapi.staticfiles import StaticFiles
from starlette.responses import FileResponse

DIST_DIR = Path(__file__).parent.parent.parent / "frontend-v2" / "dist"

def mount_spa(app: FastAPI):
    """Mount the SPA static files and catch-all."""
    if not DIST_DIR.exists():
        return  # Dev mode — Vite serves the frontend

    app.mount("/assets", StaticFiles(directory=DIST_DIR / "assets"), name="assets")

    @app.get("/{path:path}")
    async def spa_catchall(path: str):
        # Serve index.html for all non-API routes
        file = DIST_DIR / path
        if file.is_file():
            return FileResponse(file)
        return FileResponse(DIST_DIR / "index.html")
```
